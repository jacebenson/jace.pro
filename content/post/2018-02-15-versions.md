        var midVersion = gs.getProperty('mid.version').split('-')[0].toLowerCase();
        var ecma3Versions = [
          'geneva',
          'fuji',
          'eureka',
          'dublin',
          'calgary',
          'berlin'
        ];
        var ecma3 = (function (){
          for(var version in ecma3Versions){
            if(midVersion == version){
              return true;
            }
          }
          return false;
        })();

        //var currentContext = Context.getCurrentContext(),

        }
